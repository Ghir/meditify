import {
  Component,
  EventEmitter,
  Input,
  Output,
  ChangeDetectionStrategy,
} from '@angular/core';
import { INITIAL_MEDIA_STATE } from '@meditation/constants/media.constants';

// Models
import { MediaState } from '@meditation/models/media.model';
import { Meditation } from '@meditation/models/meditation.model';

@Component({
  selector: 'app-media-content',
  templateUrl: './media-content.component.html',
  styleUrls: ['./media-content.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush,
})
export class MediaContentComponent {
  @Input()
  meditation: Meditation;

  @Input()
  seekbarValue: number;

  @Input()
  media: MediaState = INITIAL_MEDIA_STATE;

  @Output()
  playAudio = new EventEmitter<void>();

  @Output()
  pauseAudio = new EventEmitter<void>();

  @Output()
  seekEnd = new EventEmitter<MouseEvent | TouchEvent>();

  constructor() {}
}

<div class="media" fxLayout="column" fxLayoutAlign="space-around center">
  <img
    class="media__image"
    *ngIf="meditation.imageUrl"
    [src]="meditation.imageUrl"
  />

  <div class="commands" fxLayout="column" fxLayoutAlign="center center">
    <div class="commands__play" fxLayoutAlign="center center">
      <ion-icon
        class="commands__icon"
        name="play"
        *ngIf="!media.playing"
        (click)="playAudio.emit()"
      ></ion-icon>

      <ion-icon
        class="commands__icon"
        name="pause"
        *ngIf="media.playing"
        (click)="pauseAudio.emit()"
      ></ion-icon>
    </div>

    <ion-range
      class="commands__range"
      name="seekbar"
      mode="md"
      min="0"
      [max]="media.durationSec"
      [value]="seekbarValue"
      (mouseup)="seekEnd.emit($event)"
      (touchend)="seekEnd.emit($event)"
    >
    </ion-range>

    <div class="commands__progress" fxLayoutAlign="space-between center">
      <span>{{ media.time }}</span>
      <span>{{ media.duration }}</span>
    </div>
  </div>
</div>
