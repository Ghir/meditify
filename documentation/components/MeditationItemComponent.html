import {
  ChangeDetectionStrategy,
  Component,
  EventEmitter,
  Input,
  Output,
} from '@angular/core';

// Models
import { content, Meditation } from '@meditation/models/meditation.model';

@Component({
  selector: 'app-meditation-item',
  templateUrl: './meditation-item.component.html',
  styleUrls: ['./meditation-item.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush,
})
export class MeditationItemComponent {
  content = content;

  @Input()
  meditation: Meditation;

  @Output()
  selected = new EventEmitter<Meditation>();

  constructor() {}
}

<ion-item (click)="selected.emit(meditation)">
  <div class="item" fxLayoutAlign="space-between center">
    <span class="item__title">{{ meditation.title }}</span>

    <div fxLayoutAlign="space-between center" fxLayoutGap="8px">
      <ion-icon
        class="item__icon"
        *ngIf="meditation[content.MEDIA]"
        name="musical-note-outline"
      ></ion-icon>

      <ion-icon
        class="item__icon"
        *ngIf="meditation.text"
        name="document-text-outline"
      ></ion-icon>
    </div>
  </div>
</ion-item>
